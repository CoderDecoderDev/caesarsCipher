// ABCDEFGHIJKLMNOPQRSTUVWXYZ      input
// NOPQRSTUVWXYZABCDEFGHIJKLM     output
const CHARACTER_MAP = {
  A: "N",
  B: "O",
  C: "P",
  D: "Q",
  E: "R",
  F: "S",
  G: "T",
  H: "U",
  I: "V",
  J: "W",
  K: "X",
  L: "Y",
  M: "Z",
  N: "A",
  O: "B",
  P: "C",
  Q: "D",
  R: "E",
  S: "F",
  T: "G",
  U: "H",
  V: "I",
  W: "J",
  X: "K",
  Y: "L",
  Z: "M",
}

function rot13(str) {
  let output = "";
  for (let i = 0; i < str.length; i++) {
    if (/[A-Z]/.test(str[i])) {
      output += CHARACTER_MAP[str[i]];
  }    
  else {
    output += str[i];
  }    
  }
  return output;
}

console.log(rot13("ABC"));
